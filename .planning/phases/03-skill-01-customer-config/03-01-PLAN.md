    return DEFAULT_CONFIG_PATH

# autoresearch/config/init.py
"""autoresearch config init — 生成 config/config.yaml 模板 (CFG-INIT-01..03).

设计 (D-02/D-03/D-06):
- 复制 config/config.example.yaml 到目标路径
- 插入 "# Generated by autoresearch config init at <ISO time>" header
- 保留所有原注释 (走文件级 shutil.copyfile, 不走 yaml 解析)
- 已存在 → exit 3 (除非 --force)
- --force 覆盖, stderr 写 "⚠ 覆盖现有 config/config.yaml"
"""
from __future__ import annotations

import datetime
import shutil
import sys
from pathlib import Path
from typing import Any

import click

from ._common import TEMPLATE_PATH, resolve_config_path


def _generate_header() -> str:
    """首行注释: 标记 generated + 下一步建议."""
    ts = datetime.datetime.now(tz=datetime.timezone.utc).isoformat(timespec="seconds")
    return (
        f"# Generated by autoresearch config init at {ts}\n"
        f"# 编辑完跑: autoresearch config validate\n"
        f"# 敏感字段不要用明文密码, 改 <env:VAR> 或 <keyring:NAME> 占位符\n"
        f"\n"
    )


def run_init(*, force: bool = False, config: str | None = None, lang: str = "zh") -> int:
    """Init 子命令主入口.

    Returns:
        exit code: 0 = 写成功, 3 = 已存在 (不覆盖), 2 = 模板缺失
    """
    target = resolve_config_path(Path(config) if config else None)

    if not TEMPLATE_PATH.exists():
        msg = (
            f"错误: 模板文件不存在 {TEMPLATE_PATH} (应在仓根随仓走)"
            if lang == "zh"
            else f"Error: template not found at {TEMPLATE_PATH}"
        )
        print(msg, file=sys.stderr)
        return 2

    if target.exists() and not force:
        msg = (
            f"配置已存在: {target} (不会覆盖). 用 --force 强制覆盖, "
            f"或用 --config <path> 写到别的路径."
            if lang == "zh"
            else f"Config already exists: {target} (not overwriting). Use --force or --config."
        )
        print(msg, file=sys.stderr)
        return 3

    # 强制覆盖警告
    if target.exists() and force:
        warn = f"⚠ 覆盖现有 {target} (--force)"
        print(warn, file=sys.stderr)

    # 确保父目录存在
    target.parent.mkdir(parents=True, exist_ok=True)

    # 1) 复制 example 模板
    shutil.copyfile(TEMPLATE_PATH, target)

    # 2) 在文件顶部插入 generated-by header (in-place 改)
    original = target.read_text(encoding="utf-8")
    target.write_text(_generate_header() + original, encoding="utf-8")

    # 3) 成功提示
    if lang == "zh":
        print(f"✅ 已生成: {target}")
        print(f"   跑: autoresearch config validate  # 校验")
        print(f"   跑: autoresearch config show      # 查看 (敏感字段脱敏)")
    else:
        print(f"✅ Generated: {target}")
        print(f"   Next: autoresearch config validate")
    return 0
```

# autoresearch/config/validate.py
"""autoresearch config validate — 校验 config/config.yaml (CFG-VAL-01..02).

设计 (D-04/D-07):
- 复用 workspace_core.config.from_path() (Phase 2 沉淀)
- 中文错误已有 (from_path 内部 _format_validation_error)
- 成功: 打印 '✅ 校验通过' + 摘要 (servers 数 / network targets 数 / wandb 启用)
- 失败: 抛 ConfigError, 中文含字段路径 + 原因 + 期望
- --json 返 JSON {ok, summary: {...}}
"""
from __future__ import annotations

import json
import sys
from pathlib import Path
from typing import Any

import click

from workspace_core.config import from_path, ConfigError
from ._common import resolve_config_path


def _summarize(cfg) -> dict[str, Any]:
    """从 Config 对象抽简短摘要 (不打印完整配置 — 留 show)."""
    return {
        "version": cfg.version,
        "servers_count": len(cfg.servers),
        "servers": [s.name for s in cfg.servers],
        "network_enabled": cfg.network.enabled,
        "network_targets_count": len(cfg.network.targets),
        "log_level": cfg.log.level,
        "wandb_enabled": cfg.wandb.enabled,
        "wandb_project": cfg.wandb.project,
    }


def run_validate(*, config: str | None = None, lang: str = "zh", as_json: bool = False) -> int:
    """Validate 子命令主入口.

    Returns:
        exit code: 0 = pass, 1 = fail (校验错误)
    """
    target = resolve_config_path(Path(config) if config else None)
    try:
        cfg = from_path(target)
    except ConfigError as e:
        if as_json:
            payload = {"ok": False, "path": str(target), "error": str(e)}
            print(json.dumps(payload, ensure_ascii=False, indent=2))
        else:
            msg = f"❌ 校验失败: {target}" if lang == "zh" else f"❌ Validation failed: {target}"
            print(msg, file=sys.stderr)
            print(str(e), file=sys.stderr)
        return 1

    summary = _summarize(cfg)
    if as_json:
        payload = {"ok": True, "path": str(target), "summary": summary}
        print(json.dumps(payload, ensure_ascii=False, indent=2))
    else:
        if lang == "zh":
            print(f"✅ {target} 校验通过")
            print(f"   version: {summary['version']}")
            print(f"   servers: {summary['servers_count']} ({', '.join(summary['servers']) or '无'})")
            print(f"   network: enabled={summary['network_enabled']}, {summary['network_targets_count']} targets")
            print(f"   log: level={summary['log_level']}")
            print(f"   wandb: enabled={summary['wandb_enabled']}, project={summary['wandb_project']}")
        else:
            print(f"✅ {target} validated")
            print(f"   version: {summary['version']}")
            print(f"   servers: {summary['servers_count']}")
            print(f"   network: enabled={summary['network_enabled']}")
    return 0
```

"""Tests for autoresearch config init (CFG-INIT-01..03)."""
import pytest
from click.testing import CliRunner

from autoresearch.cli import main
from autoresearch.config import run_init


def test_init_creates_config_yaml_with_chinese_comments():
    """CFG-INIT-01 + CFG-INIT-02: 生成的 yaml 含中文注释."""
    runner = CliRunner()
    with runner.isolated_filesystem():
        result = runner.invoke(main, ["config", "init"])
    assert result.exit_code == 0
    # 找到生成的 config/config.yaml
    import os
    p = os.path.join(os.getcwd(), "config", "config.yaml")
    assert os.path.exists(p)
    content = open(p, encoding="utf-8").read()
    # 至少含一个中文 token (注释或字段)
    assert any("\u4e00" <= c <= "\u9fff" for c in content), "no Chinese chars in generated config"
    # 必含 generated 头
    assert "Generated by autoresearch config init" in content


def test_init_exits_3_when_config_exists():
    """CFG-INIT-03: 重复 init 返 exit 3."""
    runner = CliRunner()
    with runner.isolated_filesystem():
        # 第一次
        r1 = runner.invoke(main, ["config", "init"])
        assert r1.exit_code == 0
        # 第二次应拒绝
        r2 = runner.invoke(main, ["config", "init"])
        assert r2.exit_code == 3
        combined = (r2.output or "") + (r2.stderr or "")
        assert "--force" in combined or "force" in combined.lower()


def test_init_force_overwrites_existing():
    """CFG-INIT-03: --force 覆盖."""
    runner = CliRunner()
    with runner.isolated_filesystem():
        r1 = runner.invoke(main, ["config", "init"])
        assert r1.exit_code == 0
        # 改一下文件
        p = "config/config.yaml"
        with open(p, "a", encoding="utf-8") as f:
            f.write("# user-added comment\n")
        # --force 覆盖
        r2 = runner.invoke(main, ["config", "init", "--force"])
        assert r2.exit_code == 0
        # 用户加的注释应该被冲掉
        content = open(p, encoding="utf-8").read()
        assert "user-added comment" not in content


def test_init_with_custom_config_path():
    """D-06: --config 写到别的路径."""
    runner = CliRunner()
    with runner.isolated_filesystem():
        result = runner.invoke(
            main, ["config", "init", "--config", "my/custom/config.yaml"]

"""Tests for autoresearch config validate (CFG-VAL-01..02)."""
import os
import json
import pytest
from click.testing import CliRunner

from autoresearch.cli import main


def _write_yaml(text: str) -> str:
    """helper: 写临时 yaml 到 isolated fs, 返路径."""
    import tempfile
    fd, p = tempfile.mkstemp(suffix=".yaml")
    os.write(fd, text.encode("utf-8"))
    os.close(fd)
    return p


def test_validate_passes_on_minimal_config(tmp_path, monkeypatch):
    """CFG-VAL-01: 合法 config 校验通过."""
    cfg = tmp_path / "config.yaml"
    cfg.write_text("version: 1\nservers: []\n", encoding="utf-8")
    monkeypatch.setenv("AUTORESEARCH_CONFIG", str(cfg))
    runner = CliRunner()
    result = runner.invoke(main, ["config", "validate"])
    assert result.exit_code == 0
    assert "✅" in result.output
    assert "校验通过" in result.output


def test_validate_fails_with_chinese_error_on_missing_field(tmp_path, monkeypatch):
