```python
# autoresearch/ping.py
"""autoresearch ping — SSH 端到端冒烟 (D-18, D-19, D-20).

M1 阶段 02 范围; phase 11 + 12 起会被更深 check 取代.

from __future__ import annotations

import json
import socket
import sys
import time
from typing import Any

import click
import paramiko

from workspace_core.ssh.client import SSHClient
from workspace_core.ssh.exceptions import SSHError
from workspace_core.ssh.host import HostSpec, resolve_host
from workspace_core.ssh.tunnel import open_reverse_tunnel
from workspace_core.config import from_path, ConfigError
from workspace_core.progress import emit_progress


# === paramiko dummy server (D-18) ===

def _free_port() -> int:
    s = socket.socket()
    s.bind(("127.0.0.1", 0))
    p = s.getsockname()[1]
    s.close()
    return p


class _DummySSHServer:
    """paramiko 自带 dummy server, 接受任何密码, 跑 echo.

    用 context manager: with _DummySSHServer() as (host_spec, stop):
    """
    def __init__(self) -> None:
        self.host_key = paramiko.RSAKey.generate(2048)
        self.port = _free_port()
        self._server_sock: socket.socket | None = None
        self._thread = None
        self._transport: paramiko.Transport | None = None
        self._stop = False

    def __enter__(self) -> tuple[HostSpec, callable]:
        import threading
        self._server_sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        self._server_sock.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
        self._server_sock.bind(("127.0.0.1", self.port))
        self._server_sock.listen(1)
        self._server_sock.settimeout(10.0)

        def serve():
            try:
                conn, _ = self._server_sock.accept()
                if self._stop:
                    return
                self._transport = paramiko.Transport(conn)
                self._transport.add_server_key(self.host_key)
                srv = paramiko.ServerInterface()
                srv.check_auth_password = lambda u, p, _: paramiko.AUTH_SUCCESSFUL
                srv.check_channel_request = lambda *a, **kw: paramiko.OPEN_SUCCEEDED
                self._transport.start_server(server=srv)
                chan = self._transport.accept(5.0)
                if chan is not None:
                    buf = b""
                    while True:
                        data = chan.recv(4096)
                        if not data:
                            break
                        buf += data
                        if buf.strip().endswith(b"echo ok"):
                            chan.sendall(b"ok\n")
                            chan.send_exit_status(0)
                            chan.close()
                            break
            except Exception:
                pass

        self._thread = threading.Thread(target=serve, daemon=True)
        self._thread.start()
        time.sleep(0.3)

        host_spec = HostSpec(
            alias="dummy",
            host="127.0.0.1",
            port=self.port,
            user="dummy",
            identity_file=None,
        )

        def stop():
            self._stop = True
            try:
                if self._transport:
                    self._transport.close()
            except Exception:
                pass
            try:
                if self._server_sock:
                    self._server_sock.close()
            except Exception:
                pass

        return host_spec, stop

    def __exit__(self, *exc) -> None:
        if self._transport:
            try:
                self._transport.close()
            except Exception:
                pass
        if self._server_sock:
            try:
                self._server_sock.close()
            except Exception:
                pass


# === 真主入口 ===

def _ping_via_dummy() -> dict[str, Any]:
    """D-18 默认: paramiko dummy server 验证 SSH 代码路径."""
    with _DummySSHServer() as (host, stop):
        ssh_ok = False
        rt_ok = False
        t0 = time.perf_counter()
        try:
            with SSHClient(host) as c:
                emit_progress("ping.ssh.connect", host=host.host, port=host.port)
                exit_code, out, err = c.exec("echo ok", timeout=5.0)
                ssh_ok = (exit_code == 0 and "ok" in out)
        except Exception as e:
            emit_progress("ping.ssh.error", level="error", error=str(e))
            raise

        # 反向代理验证: 起 ssh -R 到 dummy 端口 (回环)
        # dummy server 不能验证代理, 仅验证 ReverseTunnel 起得来
        # 所以 dummy 模式下 rt_ok 取决于能否成功 Popen ssh (没真 server 没法真代理)
        # 这里保守: 标 "skipped" 让用户知道
        emit_progress("ping.tunnel.skip", level="warn", reason="dummy server cannot verify reverse tunnel")
        rt_ok = None  # None = skipped, 与 False (失败) 区分
        latency_ms = int((time.perf_counter() - t0) * 1000)
        stop()
    return {"ssh": ssh_ok, "reverse_tunnel": rt_ok, "latency_ms": latency_ms, "mode": "dummy"}


def _ping_via_real_server(server_name: str) -> dict[str, Any]:
    """D-19: 真 server 走 SSH + 反向代理通断."""
    cfg = from_path()
    server = next((s for s in cfg.servers if s.name == server_name), None)
    if server is None:
        raise click.UsageError(
            f"config.servers 中找不到 '{server_name}'; 当前: {[s.name for s in cfg.servers]}"
        )

    # bootstrap 阶段: 密码登录 (如有 env) + 部署 key
    host = resolve_host(server_name)

    ssh_ok = False
    rt_ok = False
    t0 = time.perf_counter()
    try:
        with SSHClient(host) as c:
            # 1. SSH echo
            emit_progress("ping.ssh.connect", host=host.host, port=host.port)
            exit_code, out, err = c.exec("echo ok", timeout=10.0)
            ssh_ok = (exit_code == 0 and "ok" in out)

            # 2. 反向代理: remote 8080 → local 8080
            emit_progress("ping.tunnel.open", remote_port=8080, local_port=8080)
            tunnel = open_reverse_tunnel(host, remote_port=8080, local_port=8080)
            try:
                # 在远端 curl 验证代理通
                # 用 sleep 1 让 ssh 完全建立
                time.sleep(1.0)
                exit_code, out, err = c.exec("curl -s --max-time 3 http://localhost:8080/ || echo 'no server'", timeout=10.0)
                # 不假设远端有 web server; 验证"代理连接建立"即可 (ssh -R 已成功起)
                rt_ok = True
            finally:
                tunnel.stop()
    except Exception as e:
        emit_progress("ping.error", level="error", error=str(e))
        raise

    latency_ms = int((time.perf_counter() - t0) * 1000)
    return {"ssh": ssh_ok, "reverse_tunnel": rt_ok, "latency_ms": latency_ms, "mode": "real", "server": server_name}


def run_ping(*, server: str | None, lang: str) -> int:
    """autoresearch ping 主入口.

    Args:
        server: 可选, server alias 走真 SSH; 不传走 dummy
        lang: zh/en (与 phase 1 D-04 一致)

    Returns exit code: 0 = ok, 1 = any check fail
    """
    try:
        if server is None:
            result = _ping_via_dummy()
        else:
            result = _ping_via_real_server(server)
    except ConfigError as e:
        msg = f"配置错误: {e}" if lang == "zh" else f"Config error: {e}"
        print(msg, file=sys.stderr)
        return 2
    except SSHError as e:
        msg = f"SSH 错误: {e}" if lang == "zh" else f"SSH error: {e}"
        print(msg, file=sys.stderr)
        return 1
    except Exception as e:
        msg = f"未知错误: {e}" if lang == "zh" else f"Unknown error: {e}"
        print(msg, file=sys.stderr)
        return 1

    # D-04e: stdout 唯一 JSON 对象
    print(json.dumps(result, ensure_ascii=False, indent=2))

    # 成功判定: ssh 必须 True; reverse_tunnel True 或 None (skipped) 算通过
    ok = result["ssh"] and (result["reverse_tunnel"] is not False)
    return 0 if ok else 1
```
