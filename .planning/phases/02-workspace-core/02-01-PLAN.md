```python
# workspace-core/ssh/host.py
from __future__ import annotations
from dataclasses import dataclass
from pathlib import Path
import paramiko

@dataclass(frozen=True)
class HostSpec:
    """解析后的目标主机规格."""
    alias: str | None      # 原始 alias (如果是从 ssh_config 解析的)
    host: str              # 实际 hostname 或 IP
    port: int              # SSH 端口 (默认 22)
    user: str              # 用户名
    identity_file: Path | None  # IdentityFile 路径 (可能 None)

def resolve_host(target: str) -> HostSpec:
    """解析目标字符串为 HostSpec.

    1. 包含 '@' → 解析为 user@host[:port] 直接形式
    2. 否则 → 当作 alias, 查 ~/.ssh/config
    """
    if "@" in target:
        return _parse_direct(target)
    return _parse_ssh_config(target)


def _parse_direct(target: str) -> HostSpec:
    """解析 user@host[:port] 形式."""
    user, rest = target.split("@", 1)
    if ":" in rest:
        host, port_s = rest.rsplit(":", 1)
        return HostSpec(alias=None, host=host, port=int(port_s), user=user, identity_file=None)
    return HostSpec(alias=None, host=rest, port=22, user=user, identity_file=None)


def _parse_ssh_config(alias: str) -> HostSpec:
    """从 ~/.ssh/config 解析 alias."""
    config_path = Path.home() / ".ssh" / "config"
    if not config_path.exists():
        raise HostResolveError(f"未找到 ssh config: {config_path}; alias='{alias}' 仅支持直接 user@host 形式")

    ssh_config = paramiko.SSHConfig.from_file(str(config_path))
    conf = ssh_config.lookup(alias)
    if not conf or "hostname" not in conf:
        raise HostResolveError(f"ssh config 中找不到 alias='{alias}' (config={config_path})")

    host = conf["hostname"]
    port = int(conf.get("port", 22))
    user = conf.get("user") or _current_user()
    id_file = None
    if "identityfile" in conf:
        # paramiko 可能返回 list
        ids = conf["identityfile"]
        id_file = Path(ids[0]) if isinstance(ids, list) else Path(ids)

    return HostSpec(
        alias=alias, host=host, port=port, user=user, identity_file=id_file
    )


def _current_user() -> str:
    import getpass
    return getpass.getuser()
```

- IdentityFile 可能是 list, 取第一个

```python
# workspace-core/ssh/keys.py
from __future__ import annotations

import os
import subprocess
from pathlib import Path

import paramiko

from .client import SSHClient
from .exceptions import KeyDeployError


SSH_KEYS_DIR = Path.home() / ".autoresearch" / "ssh_keys"


def ensure_local_keypair() -> tuple[Path, Path]:
    """确保本机有 ed25519 keypair; 没有就生成.

    Returns (private_key_path, public_key_path).
    """
    SSH_KEYS_DIR.mkdir(parents=True, exist_ok=True)
    priv = SSH_KEYS_DIR / "id_ed25519"
    pub = SSH_KEYS_DIR / "id_ed25519.pub"

    if not priv.exists():
        # 用 ssh-keygen 生成
        subprocess.run(
            ["ssh-keygen", "-t", "ed25519", "-N", "", "-f", str(priv), "-C", "autoresearch"],
            check=True,
            capture_output=True,
        )
    return priv, pub


def public_key_fingerprint(pub_path: Path) -> str:
    """算公钥 fingerprint (用作已部署标记)."""
    result = subprocess.run(
        ["ssh-keygen", "-lf", str(pub_path)],
        check=True,
        capture_output=True,
        text=True,
    )
    return result.stdout.split()[1]  # "256 SHA256:xxx ..." -> SHA256:xxx


def _deployed_marker_path(host_alias: str) -> Path:
    """~/.autoresearch/ssh_keys/<host>.deployed 标记文件路径."""
    safe = host_alias.replace("/", "_").replace("@", "_")
    return SSH_KEYS_DIR / f"{safe}.deployed"


def is_key_deployed(host_alias: str, pub_path: Path) -> bool:
    """检查本机公钥是否已部署到 <host> (依据标记文件)."""
    marker = _deployed_marker_path(host_alias)
    if not marker.exists():
        return False
    return marker.read_text().strip() == public_key_fingerprint(pub_path)


def mark_key_deployed(host_alias: str, pub_path: Path) -> None:
    """标记 <host> 已部署本机公钥."""
    marker = _deployed_marker_path(host_alias)
    marker.write_text(public_key_fingerprint(pub_path))


def deploy_public_key(client: SSHClient, host_alias: str, password: str) -> bool:
    """Bootstrap 阶段: 用密码登录后部署本机公钥到远端.

    Returns True 如果是新部署 (之前没部署过); False 如果已部署.
    """
    _, pub_path = ensure_local_keypair()

    if is_key_deployed(host_alias, pub_path):
        return False  # 之前已部署, 跳过

    pub_key_content = pub_path.read_text().strip()
    sftp = client.sftp()
    try:
        # 读远端 authorized_keys, 追加, 写回
        remote_ak = ".ssh/authorized_keys"
        try:
            existing = sftp.open(remote_ak, "r").read().decode("utf-8", errors="ignore")
        except IOError:
            existing = ""

        if pub_key_content in existing:
            # 远端已有 (但本机标记丢了), 仅补标记
            mark_key_deployed(host_alias, pub_path)
            return False

        with sftp.open(remote_ak, "a") as f:
            f.write(pub_key_content + "\n")

        # 权限收尾
        sftp.chmod(remote_ak, 0o600)
        sftp.chmod(".ssh", 0o700)
    except Exception as e:
        raise KeyDeployError(
            f"无法部署公钥到 {host_alias}: {e}"
        ) from e
    finally:
        sftp.close()

    mark_key_deployed(host_alias, pub_path)
    return True
```

注意：
- `ensure_local_keypair` 用 `ssh-keygen` 生成 (不依赖 paramiko RSAKey; 跨平台走系统命令)
- `is_key_deployed` 用 fingerprint (SHA256:xxx) 而不是整个公钥 (公钥可能含注释变化)
- `deploy_public_key` 用 sftp 追加 authorized_keys，幂等

```python
# workspace-core/ssh/client.py
from __future__ import annotations

import os
import time
from contextlib import contextmanager

import paramiko

from .exceptions import AuthError, ConnectError, SSHError
from .host import HostSpec, resolve_host
from . import keys as ssh_keys


DEFAULT_CONNECT_TIMEOUT_S = 5.0
DEFAULT_CMD_TIMEOUT_S = 30.0
DEFAULT_RETRIES = 3
BACKOFF_BASE_S = 1.0


class SSHClient:
    """paramiko 抹平抽象 (D-01).

    8 skill 只用本接口, 不直接 import paramiko.
    """

    def __init__(self, host: HostSpec) -> None:
        self.host = host
        self._client: paramiko.SSHClient | None = None

    # ===== 登录 / 注销 =====

    def connect(
        self,
        *,
        connect_timeout: float = DEFAULT_CONNECT_TIMEOUT_S,
        retries: int = DEFAULT_RETRIES,
    ) -> None:
        """连接 + 认证. 支持 3 阶段凭据解析 (D-03):
        1. ssh-agent (key)
        2. IdentityFile (key) — 从 ssh_config 或 env SSH_KEYFILE_<host>
        3. password (env SSH_PASSWORD_<host>) — 仅 bootstrap 用
        """
        if self._client is not None:
            return  # 幂等

        # Bootstrap 阶段: 用密码登录
        bootstrap_password = os.environ.get(f"SSH_PASSWORD_{self.host.alias.upper()}") if self.host.alias else None

        # 优先用 key (尝试 IdentityFile / ssh-agent / 默认位置)
        last_err: Exception | None = None
        for attempt in range(retries + 1):
            if attempt > 0:
                time.sleep(BACKOFF_BASE_S * (2 ** (attempt - 1)))
            try:
                client = paramiko.SSHClient()
                client.load_system_host_keys()
                client.set_missing_host_key_policy(paramiko.WarningPolicy)

                connect_kwargs: dict = {
                    "hostname": self.host.host,
                    "port": self.host.port,
                    "username": self.host.user,
                    "timeout": connect_timeout,
                    "allow_agent": True,
                    "look_for_keys": True,
                }
                if self.host.identity_file:
                    connect_kwargs["key_filename"] = str(self.host.identity_file)

                if bootstrap_password:
                    connect_kwargs["password"] = bootstrap_password

                client.connect(**connect_kwargs)
                self._client = client
                return
            except paramiko.AuthenticationException as e:
                # Auth 失败 = 立刻报错, 不重试
                raise AuthError(
                    f"认证失败 host={self.host.host}:{self.host.port} user={self.host.user}; "
                    f"如需密码引导, 设置 env SSH_PASSWORD_<ALIAS>"
                ) from e
            except (paramiko.SSHException, OSError) as e:
                last_err = ConnectError(
                    f"连接失败 host={self.host.host}:{self.host.port} (attempt={attempt + 1}/{retries + 1}): {e}"
                )

        raise last_err or ConnectError(f"连接失败 host={self.host.host}:{self.host.port}")

    def close(self) -> None:
        if self._client is not None:
            self._client.close()
            self._client = None

    def __enter__(self) -> "SSHClient":
        self.connect()
        return self

    def __exit__(self, *exc) -> None:
        self.close()

    # ===== 命令执行 =====

    def exec(
        self,
        command: str,
        *,
        timeout: float = DEFAULT_CMD_TIMEOUT_S,
    ) -> tuple[int, str, str]:
        """执行命令, 返回 (exit_code, stdout, stderr)."""
        if self._client is None:
            raise SSHError(f"未连接; 先调 connect() — host={self.host.host}")
        stdin, stdout, stderr = self._client.exec_command(command, timeout=timeout)
        exit_code = stdout.channel.recv_exit_status()
        return (
            exit_code,
            stdout.read().decode("utf-8", errors="replace"),
            stderr.read().decode("utf-8", errors="replace"),
        )

    # ===== SFTP =====

    def sftp(self) -> paramiko.SFTPClient:
        if self._client is None:
            raise SSHError(f"未连接; 先调 connect() — host={self.host.host}")
        return self._client.open_sftp()

    # ===== Bootstrap helper =====

    def bootstrap(self) -> bool:
        """Bootstrap-then-Key 模式 (D-03): 密码登录 → 部署公钥 → 标完成.

        Returns True = 这次部署了新公钥, False = 已部署过.
        """
        if not self.host.alias:
            raise SSHError("bootstrap 仅适用于 alias 形式; user@host 直连无法标记")

        # 必须先 connect (可能用密码)
        if self._client is None:
            self.connect()

        password = os.environ.get(f"SSH_PASSWORD_{self.host.alias.upper()}")
        if not password:
            raise SSHError(
                f"需要 env SSH_PASSWORD_{self.host.alias.upper()} 提供一次性密码"
            )

        return ssh_keys.deploy_public_key(self, self.host.alias, password)
```

注意：
- `connect` 内 3 阶段凭据: ssh-agent → IdentityFile → 密码 (env)
- Auth 错误不重试, 网络错误重试 3 次
- `bootstrap()` 是高层 API，编排 connect + deploy
- `__enter__` / `__exit__` 支持 `with SSHClient(host) as c:` 语法

```python
# workspace-core/ssh/tunnel.py
from __future__ import annotations

import shlex
import signal
import subprocess
import time
from dataclasses import dataclass
from pathlib import Path

from .exceptions import TunnelError
from .host import HostSpec


@dataclass
class ReverseTunnel:
    """ssh -R 反向代理的运行实例."""

    pid: int
    host_alias: str
    remote_port: int
    local_port: int
    log_path: Path

    def stop(self, timeout_s: float = 5.0) -> None:
        """优雅停止 tunnel 进程."""
        try:
            self.log_path.parent.mkdir(parents=True, exist_ok=True)
        except OSError:
            pass
        try:
            self.proc.wait(timeout_s)
        except (subprocess.TimeoutExpired, AttributeError):
            self.proc.terminate()
            try:
                self.proc.wait(timeout_s)
            except subprocess.TimeoutExpired:
                self.proc.kill()


def open_reverse_tunnel(
    host: HostSpec,
    *,
    remote_port: int,
    local_port: int,
    identity_file: Path | None = None,
    log_dir: Path | None = None,
) -> ReverseTunnel:
    """起一条 ssh -R 反向代理, 返回 ReverseTunnel 实例.

    命令模板:
        ssh -N -R <remote_port>:localhost:<local_port> \\
           -o ServerAliveInterval=30 -o ServerAliveCountMax=3 \\
           -o ExitOnForwardFailure=yes \\
           [-i <identity_file>] \\
           <user>@<host>
    """
    if log_dir is None:
        log_dir = Path.home() / ".autoresearch" / "logs"
    log_dir.mkdir(parents=True, exist_ok=True)
    log_path = log_dir / f"tunnel-{host.alias or host.host}.log"

    cmd = [
        "ssh", "-N",
        "-R", f"{remote_port}:localhost:{local_port}",
        "-o", "ServerAliveInterval=30",
        "-o", "ServerAliveCountMax=3",
        "-o", "ExitOnForwardFailure=yes",
        "-o", "StrictHostKeyChecking=accept-new",
        "-o", f"UserKnownHostsFile={Path.home() / '.ssh' / 'known_hosts'}",
    ]
    if identity_file:
        cmd += ["-i", str(identity_file)]
    cmd += [f"{host.user}@{host.host}"]

    try:
        log_file = open(log_path, "a")
        proc = subprocess.Popen(
            cmd,
            stdout=log_file,
            stderr=subprocess.STDOUT,
            start_new_session=True,  # 独立进程组, 避免 SIGINT 传到子进程
        )
    except FileNotFoundError:
        raise TunnelError("找不到 ssh 命令; 请确认 PATH 含 /usr/bin/ssh")
    except OSError as e:
        raise TunnelError(f"无法启动 ssh tunnel: {e}") from e

    # 短延迟看是否立即死 (ExitOnForwardFailure=yes 时, 失败立即退出)
    time.sleep(0.5)
    if proc.poll() is not None:
        log_tail = log_path.read_text(errors="ignore")[-500:]
        raise TunnelError(
            f"ssh tunnel 启动后立即退出 (rc={proc.returncode}); log tail:\n{log_tail}"
        )

    return ReverseTunnel(
        pid=proc.pid,
        host_alias=host.alias or host.host,
        remote_port=remote_port,
        local_port=local_port,
        log_path=log_path,
    )
```

注意：
- 用 `subprocess.Popen` + `start_new_session=True` 避免 SIGINT 串到子进程
- `ExitOnForwardFailure=yes` 让失败立即退出 (不要静默挂)
- `StrictHostKeyChecking=accept-new` 自动接受新 host key
- log 写到 `~/.autoresearch/logs/tunnel-<host>.log`
